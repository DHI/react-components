# useSessionStorage

The `useSessionStorage` hook leverages the [sessionStorage](https://developer.mozilla.org/en-US/docs/Web/API/Window/sessionStorage) getter and setter functions, in a React friendly way.

## Example

<Story name="useSessionStorage">
  <Canvas>
    <SessionStorageExample />
  </Canvas>
  <Syntax code={`
    import { FC } from "react";
    import { useSessionStorage } from "@dhi/react-components-lab";\n
    const App: FC = () => {
      const [showData, setShowData] = useSessionStorage<boolean | undefined>('showTheData');\n
      return (
        <div>
          {\`showTheData: \${String(showData)}\`}
          <button onClick={() => setShowData(!Boolean(showData))}>
            Toggle "showTheData" session storage value
          </button>
        </div>
      )
    }
  `}/>
</Story>
